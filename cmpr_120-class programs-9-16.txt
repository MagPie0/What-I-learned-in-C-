// ===============================
// Declare a variable of int data type
// ===============================
#include <iostream>
using namespace std;

int main()
{
	// Variable declaration
	int age;	

	cout << "How old are you?  ";
	cin >> age;

	cout << "You don't look " << age << " years old.\n\n";

	// This is the assignment operator = 
	age = 33;		

	cout << "So now you are " << age << " years old.\n\n";

	system("pause");
	return 0;
}

/* OUTPUT
How old are you?  22
You don't look 22 years old.

So now you are 33 years old.

Press any key to continue . . .*/
